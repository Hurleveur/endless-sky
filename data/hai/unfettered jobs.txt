# make the amount random
mission "Unfettered Silicon Asteroid Mining"
	name `Gather silicon asteroids`
	job
	repeat
	description `Mine and then collect 50 tons of Silicon from the asteroid field in Wah Yoot and then bring them back to the planet for <payment>.`
	source "Darkcloak"
	to offer
		has "First Contact: Unfettered: declined"
		random < 70
	on complete
		payment 250000
		dialog `The Unfettered seem happy. "This will allow us to repair our ships !" They hand you <payment> credits.`
		outfit "Silicon" -50


# scripted version of this one for friend
mission "Unfettered Medicals"
	name "Hai medicals to <planet>"
	job
	repeat
	deadline 20
	description "The Unfettered hospitals on Darkcloak are in urgent need of resupply. Collect a supply of <tons> of medicals from <stopovers>, and bring it to the Unfettered on <destination> by <date> for <payment>."
	source "Darkcloak"
	to offer
		has "Unfettered Jump Drive 1: offered"
		random < 80
	cargo "medicals" 15 2 .05
	stopover
		distance 5 7
		attributes "hai"
		attributes "spaceport"
	on visit
		dialog phrase "generic cargo on visit"
	on stopover
		dialog phrase `You are surprised to see the Hai are more than happy to give you the medical supplies needed. "Tell our brothers are sisters to stay strong,"	they say to you.`
	on complete
		payment 10000 3000
		dialog `The local health specialists seems relieved to learn that you have managed to bring the medical supplies back in time.`



mission "Unfettered Challenge"
	name `Defeat the young warlord`
	job
	repeat
	description `A young Unfettered warlord has provoked you in a duel, disable his ship for <payment>.`
	source
		attributes "unfettered"
	to offer
		has "Unfettered Jump Drive 2: offered"
		random < 70
	npc save disable
		government "Hai (Unfettered Challenge)"
		personality disables staying heroic target
		fleet
			names "hai"
			variant
				"Lightning Bug (Pulse)" 3
			dialog `Well done captain ! But I will not be so easy to beat next time.`
	on complete
		payment 250000
		dialog `You receive a message from the <npc> : "Here is the agreed <payment>, captain. You deserved it."`
